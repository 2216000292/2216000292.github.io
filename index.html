class Solution:
def mergeTwoLists(self, l1: ListNode, l2: ListNode) -> ListNode:
    if l1 is None:
        return l2
    elif l2 is None:
        return l1
    elif l1.val < l2.val:
        l1.next = self.mergeTwoLists(l1.next, l2)
        return l1
    else:
        l2.next = self.mergeTwoLists(l1, l2.next)
        return l2

lc20 有效的括号 Valid Parentheses

个人解析：初始化一个哈希pairs，依次遍历字符串，如果遇到左括号则放入栈，如果遇到右括号则跟栈顶对比利用pairs，如果两者匹配则pop(),如果不匹配或者栈顶已经为空则return false，同时在function前面添加一点常识字符串若为单数不符合题目要求，减少运算时间。

附：stack[-1] 表示的是栈顶值

时间复杂度为： O(n)

class Solution:
def isValid(self, s: str) -> bool:
    if len(s) % 2 == 1:     #如果长度不为偶数则return false
        return False
    
    pairs = {       #初始化哈希表
        ")": "(",
        "]": "[",
        "}": "{",
    }
    stack = list()
    for ch in s:
        if ch in pairs:
            if not stack or stack[-1] != pairs[ch]:   
                return False
            stack.pop()
        else:
            stack.append(ch)
    
    return not stack
